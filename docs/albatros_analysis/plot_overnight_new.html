albatros_analysis.plot_overnight_new

A script to plot direct spectra products over several hours (combining multiple files). It will generate a single plot showing what the data looks like for an entire night's (or several nights') run. It takes a start timestamp and end timestamp and automatically finds all files that lie between these timestamps. (Timestamp format YYYYMMDD_hhmmss)

Needs a top-level directory that contains sub-directories with "short" 5 digit timestamps. (Each of these sub-dirs has sub-dirs with full 10 digit timestamps.) E.g.

~/data_auto_cross/snap1/

In a particular direct data file, each data point (for, say, pol00) is averaged over roughly 6 seconds by the FPGA. This may be too high a resolution if you're plotting for several hourse. Use -a [avglen] to specify how many points to average over.

python plot_overnight_new.py ~/data_auto_cross/snap3/ 20210730_000000 20210730_070000 -o ~/outputs/ -a 50 -l

The above example calculates direct data products averaged over 5 minuts (50 points) for data collected between Jul-30 12 AM to Jul-30 7 AM. And once again, use -l for logscale plots.

It is the user's responsibility to make sure that a contiguous chunk of data is available between the two timestamps. If there's some discontinuity (e.g. data wasn't collected for 2 hours in the 7 hour window), it'll be obvious from the plots. In this sense, this is a useful diagnostic tool.

 51def get_data_arrs(data_dir: str, ctime_start: str, ctime_stop: str, chunk_time, blocklen, mytz):
 52    """
 53    Given the path to a Big data directory (i.e. directory contains the directories
 54    labeled by the first 5 digits of the ctime date), gets all the data in some time interval.
 55
 56    Parameters:
 57    -----------
 58
 59    data_dir: str
 60        path to data directory
 61
 62    ctime_start, ctime_stop: str
 63        desired start and stop time in ctime
 64
 65    chunk_time: ??
 66    
 67    blocklen: (probably int)
 68    
 69    mytz: (what kind of object is this? probably pytz.tzfile)
 70        Timezone (of the dish at collection?)
 71
 72    Returns:
 73    --------
 74
 75    cimte_start, ctime_stop: int
 76        start and stop times in ctime
 77
 78    pol00,pol11,pol01r,pol01i: array
 79        2D arrays containing the data for given time interval for autospectra
 80        as well as cross spectra. pol00 corresponds to adc0 and pol11 to adc3
 81    """
 82    print("\n################### READING DATA ###################")
 83    print(f"Files requested between timestamps {ctime_start} to {ctime_stop}")
 84    print(
 85        f"Corresponding UTC time: {datetime.utcfromtimestamp(ctime_start)} to {datetime.utcfromtimestamp(ctime_stop)}"
 86    )
 87
 88    # all the dirs between the timestamps. read all, append, average over chunk length
 89    data_subdirs = sft.time2fnames(ctime_start, ctime_stop, data_dir)
 90    print("total data subdirs", len(data_subdirs))
 91    print("First and last subdirs:", data_subdirs[0], data_subdirs[-1])
 92    data_subdirs.sort()
 93
 94    if len(data_subdirs) == 0:
 95        print("NOTHING WAS READ. CHECK TSTAMPS")
 96        sys.exit(1)
 97
 98    # rough estimate of number of rows we'll read
 99    nrows_guess = len(data_subdirs) * ((int(3600 / chunk_time / blocklen) + 1) + 1)
100    # print("Starting with a guess of ", nrows_guess)
101    print("guessed rows", nrows_guess)
102    pol00 = np.zeros((nrows_guess + 500, 2048))
103
104    nrows = 0
105
106    t1 = time.time()
107    new_dirs = [d + "/pol00.scio.bz2" for d in data_subdirs]
108    datpol00 = scio.read_files(new_dirs)
109    new_dirs = [d + "/pol11.scio.bz2" for d in data_subdirs]
110    datpol11 = scio.read_files(new_dirs)
111    new_dirs = [d + "/pol01r.scio.bz2" for d in data_subdirs]
112    datpol01r = scio.read_files(new_dirs)
113    new_dirs = [d + "/pol01i.scio.bz2" for d in data_subdirs]
114    datpol01i = scio.read_files(new_dirs)
115    print(time.time() - t1, f"Read {len(data_subdirs)} files")
116
117    # average everything if blocklen>1
118    # print("first row must be in:", data_subdirs[0])
119
120    myavgfunc = partial(get_avg, block=blocklen)
121    if blocklen > 1:
122        t1 = time.time()
123        with Pool(os.cpu_count()) as p:
124            avgpol00 = p.map(myavgfunc, datpol00)
125            avgpol11 = p.map(myavgfunc, datpol11)
126            avgpol01r = p.map(myavgfunc, datpol01r)
127            avgpol01i = p.map(myavgfunc, datpol01i)
128        print(time.time() - t1, "averaged everything")
129    else:
130        avgpol00 = datpol00
131        avgpol11 = datpol11
132        avgpol01r = datpol01r
133        avgpol01i = datpol01i
134
135    print(len(avgpol00))
136    t1 = time.time()
137    tstart = 0
138    tend = 0
139    for i, d in enumerate(avgpol00):
140        # print("Mean, median are", np.mean(d,axis=0),np.median(d,axis=0))
141        print("working on", data_subdirs[i])
142        if d is None:
143            continue
144        if i == 0:
145            pol00[: d.shape[0]] = d
146            nrows += d.shape[0]
147            ts = get_ts_from_name(data_subdirs[i])
148
149            tstart = ts  # save starting time for user output
150            ts = ts + d.shape[0] * chunk_time * blocklen
151            continue
152        newts = get_ts_from_name(data_subdirs[i])
153        diff = int((newts - ts) / chunk_time / blocklen)
154        # each cell in the plot represents a minimum time of blocklen * chunktime.
155        # That's the time resolution for the plot. Can't catch gaps < resolution.
156        if diff > 0:
157            print(f"significant diff b/w files {tstart} and {newts} of:", diff, "rows")
158            pol00[nrows : nrows + diff, :] = np.nan
159            pol00 = np.append(pol00, np.zeros((diff, 2048)), axis=0)
160            nrows += diff
161        # print(nrows, d.shape)
162        # print(nrows,nrows+d.shape[0],pol00.shape,"heh")
163        pol00[nrows : nrows + d.shape[0], :] = d
164        nrows += d.shape[0]
165        # print("reading", data_subdirs[i], "with size ", d.shape[0], "NROWS", oldnrows,nrows)
166        tstart = newts
167        ts = newts + d.shape[0] * chunk_time * blocklen
168    tend = ts
169    # print("HERE")
170    # once we have pol00, we know the exact size. use it
171    tstart = get_ts_from_name(
172        data_subdirs[0]
173    )  # tstart was replaced above for missing gap info
174    print("############################################################")
175    print(
176        f"First file at: {tstart}, Last file at: {get_ts_from_name(data_subdirs[-1])}"
177    )
178    print(f"Plotting all data starting {tstart} and ending {int(tend)}")
179    ts, te = list(map(partial(get_localtime_from_UTC, mytz=mytz), [tstart, tend]))
180    print(
181        f"In Local time: {ts.strftime('%b-%d %H:%M:%S')} to {te.strftime('%b-%d %H:%M:%S')} in {mytz.zone}"
182    )
183    print("Final nrows:", nrows)
184
185    pol00 = pol00[:nrows].copy()
186    # print(pol00.shape)
187
188    pol11 = np.zeros((nrows, 2048))
189    pol01r = np.zeros((nrows, 2048))
190    pol01i = np.zeros((nrows, 2048))
191
192    nrows = 0
193    for i in range(len(avgpol00)):
194        if (avgpol11[i] is None) or (avgpol01i[i] is None) or (avgpol01r[i] is None):
195            continue
196        if i == 0:
197            r = avgpol11[i].shape[0]
198            pol11[:r, :] = avgpol11[i]
199            pol01r[:r, :] = avgpol01r[i]
200            pol01i[:r, :] = avgpol01i[i]
201            nrows += r
202            ts = get_ts_from_name(data_subdirs[i]) + r * chunk_time * blocklen
203            continue
204        newts = get_ts_from_name(data_subdirs[i])
205        diff = int((newts - ts) / chunk_time / blocklen)
206        if diff > 0:
207            pol11[nrows : nrows + diff, :] = np.nan
208            pol01r[nrows : nrows + diff, :] = np.nan
209            pol01i[nrows : nrows + diff, :] = np.nan
210            nrows += diff
211        r = avgpol11[i].shape[0]
212        pol11[nrows : nrows + r, :] = avgpol11[i]
213        pol01r[nrows : nrows + r, :] = avgpol01r[i]
214        pol01i[nrows : nrows + r, :] = avgpol01i[i]
215        nrows += r
216        ts = newts + r * chunk_time * blocklen
217
218    t2 = time.time()
219    # print('Time taken to concatenate data:',t2-t1)
220    # print("pol00, pol11,pol01r, pol01i shape:", pol00.shape,pol11.shape,pol01r.shape,pol01i.shape)
221    pol00 = np.ma.masked_invalid(pol00)
222    pol11 = np.ma.masked_invalid(pol11)
223    pol01r = np.ma.masked_invalid(pol01r)
224    pol01i = np.ma.masked_invalid(pol01i)
225    return pol00, pol11, pol01r, pol01i, tstart, tend

228def get_avg(arr, block=10):
229    """
230    Fast average array over a given block size.
231    """
232    if arr is None:
233        return None
234    iters = arr.shape[0] // block
235    leftover = arr.shape[0] % block
236    # print(iters,leftover)
237    nrows = iters + int(leftover > 0)
238    ncols = arr.shape[1]
239    newarr = np.zeros((nrows, ncols), dtype=arr.dtype)
240    cmp1 = np.median(arr[0, :]) / np.median(
241        arr[1, :]
242    )  # temporary fix, assuming only first row is expected to be faulty
243    if cmp1 > 1e2:
244        # skip first row before averaging for first block
245        newarr[0, :] = np.mean(arr[1:block, :], axis=0)
246        newarr[1:iters, :] = np.mean(
247            arr[block : iters * block, :].reshape(-1, block, ncols), axis=1
248        )
249    else:
250        # print(f"Shape of passed arr {arr.shape} and shape of new arr {newarr.shape}")
251        newarr[:iters, :] = np.mean(
252            arr[: iters * block, :].reshape(-1, block, ncols), axis=1
253        )
254    if leftover:
255        newarr[iters, :] = np.mean(arr[iters * block :, :], axis=0)
256    return newarr

259def get_stats(data_arr):
260    """
261    Given a 2D array containing some data chunk, returns the
262    min, median, mean, and max over that chunk.
263    """
264    # print("WHERE MIN ZERO",np.where(np.min(data_arr,axis=0)==0))
265    # print("WHERE MEDIAN ZERO",np.where(np.median(data_arr,axis=0)==0))
266    # print("MEDIAN",np.median(data_arr,axis=0))
267    # print("MEDIAN MA",np.ma.median(data_arr,axis=0))
268    if logplot:
269        stats = {
270            "min": np.log10(np.ma.min(data_arr, axis=0)),
271            "median": np.log10(np.ma.median(data_arr, axis=0)),
272            "mean": np.log10(np.ma.mean(data_arr, axis=0)),
273            "max": np.log10(np.ma.max(data_arr, axis=0)),
274        }
275    else:
276        stats = {
277            "min": np.ma.min(data_arr, axis=0),
278            "median": np.ma.median(data_arr, axis=0),
279            "mean": np.ma.mean(data_arr, axis=0),
280            "max": np.ma.max(data_arr, axis=0),
281        }
282    return stats

285def get_vmin_vmax(data_arr):
286    """
287    Automatically gets vmin and vmax for colorbar
288    """
289    # print("shape of passed array", data_arr.shape, data_arr.dtype)
290    xx = data_arr[~data_arr.mask].data
291    med = np.percentile(xx, 50)
292    # print(med, "median")
293    u = np.percentile(xx, 99)
294    b = np.percentile(xx, 1)
295    xx_clean = xx[(xx <= u) & (xx >= b)]  # remove some outliers for better plotting
296    stddev = np.std(xx_clean)
297    vmin = max(med - 2 * stddev, 10**7)
298    vmax = med + 2 * stddev
299    # print("vmin, vmax are", vmin, vmax)
300    return vmin, vmax

303def get_ylim_times(t_i, t_f):
304    """
305    Gets the y limits in matplotlib's date format for a given initial time
306    and final time. t_i and t_f must be given in ctime
307    """
308    # getlocaltime = lambda tstamp: datetime.fromtimestamp(int(tstamp),tz=pytz.utc).astimezone(tz=mytz)
309    y_lims = list(map(datetime.utcfromtimestamp, [t_i, t_f]))
310    y_lims_plt = mdates.date2num(y_lims)
311    # date2num is NOT tz aware.
312    # will return same value regardless of tz of passed datetime object.
313    # pass tz to formatter and tick locators
314    return y_lims_plt

def get_avg(arr, block=10): 
229def get_avg(arr, block=10):
230    """
231    Fast average array over a given block size.
232    """
233    if arr is None:
234        return None
235    iters = arr.shape[0] // block
236    leftover = arr.shape[0] % block
237    # print(iters,leftover)
238    nrows = iters + int(leftover > 0)
239    ncols = arr.shape[1]
240    newarr = np.zeros((nrows, ncols), dtype=arr.dtype)
241    cmp1 = np.median(arr[0, :]) / np.median(
242        arr[1, :]
243    )  # temporary fix, assuming only first row is expected to be faulty
244    if cmp1 > 1e2:
245        # skip first row before averaging for first block
246        newarr[0, :] = np.mean(arr[1:block, :], axis=0)
247        newarr[1:iters, :] = np.mean(
248            arr[block : iters * block, :].reshape(-1, block, ncols), axis=1
249        )
250    else:
251        # print(f"Shape of passed arr {arr.shape} and shape of new arr {newarr.shape}")
252        newarr[:iters, :] = np.mean(
253            arr[: iters * block, :].reshape(-1, block, ncols), axis=1
254        )
255    if leftover:
256        newarr[iters, :] = np.mean(arr[iters * block :, :], axis=0)
257    return newarr
